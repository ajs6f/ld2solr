3   import static com.google.common.collect.Collections2.transform;
4   
5   import java.util.Collection;
6   import java.util.HashMap;
7   import java.util.Map;
8   
9   import com.google.common.base.Function;
10  import com.google.common.collect.Maps;
11  import com.google.common.collect.Maps.EntryTransformer;
12  
13  /**
14   * A {@link Map} of named fields. Any field may be multivalued.
15   * 
16   * @author ajs6f
17   * 
18   */
19  public class NamedFields extends HashMap<String, Collection<String>> {
20  
21  	private static final long serialVersionUID = 1L;
22  
23  	public static final String ID_FIELD = "id";
24  
25  	private String id;
26  
27  	public NamedFields(final Map<String, Collection<?>> wrappedFields) {
28  		this.putAll(Maps.transformEntries(wrappedFields, asEntryOfStringToCollectionOfStrings));
29  	}
30  
31  	/**
32  	 * @return a identifer for this record
33  	 */
34  	public String id() {
35  		return id != null ? id : get(ID_FIELD).iterator().next();
36  	}
37  
38  	/**
39  	 * @param id
40  	 * @return this {@link NamedFields} for continued operation
41  	 */
42  	public NamedFields id(final String id) {
43  		this.id = id;
44  		return this;
45  	}
46  
47  	private static final Function<Collection<?>, Collection<String>> asCollectionOfStrings = new Function<Collection<?>, Collection<String>>() {
48  
49  		@Override
50  		public Collection<String> apply(final Collection<?> input) {
51  			return transform(input, asString);
52  		}
53  	};
54  
55  	private static final EntryTransformer<String, Collection<?>, Collection<String>> asEntryOfStringToCollectionOfStrings = new EntryTransformer<String, Collection<?>, Collection<String>>() {
56  
57  		@Override
58  		public Collection<String> transformEntry(final String key, final Collection<?> value) {
59  
60  			return asCollectionOfStrings.apply(value);
61  		}
62  	};
63  	private static final Function<Object, String> asString = new Function<Object, String>() {
64  
65  		@Override
66  		public String apply(final Object input) {
67  			return input.toString();
68  		}
69  	};
70  
71  }
