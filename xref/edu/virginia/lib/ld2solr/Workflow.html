51  /**
52   * Assembles and operates an indexing workflow from SPI implementations, as well
53   * as CLI support therefor.
54   * 
55   * @author ajs6f
56   * 
57   */
58  public class Workflow {

60  	/*
61  	 * because LDPath uses java.util.ServiceLoader to search the context
62  	 * classpath, we need to explicitly call out LDPath here to make sure it
63  	 * gets initialized properly
64  	 */

66  	private final LDPath<String> testTransform = new LDPath<>(null);

76  	private Integer numIndexerThreads = ThreadedStage.DEFAULT_NUM_THREADS;

78  	private static final Logger log = getLogger(Workflow.class);
79  
80  	public Set<Resource> cache(final Set<Resource> uris) throws InterruptedException {
81  		final Set<Resource> successfullyRetrieved = cacheAssembler.call();
82  		final Set<Resource> failures = difference(uris, successfullyRetrieved);
83  		if (failures.size() > 0) {
84  			log.error("Failed to retrieve some resources!");
85  			for (final Resource failure : failures) {
86  				log.warn("Resource: {} could not be retrieved!", failure);
87  			}
88  		}
89  		cacheAssembler.shutdown();
90  		return successfullyRetrieved;
91  	}
92  
93  	public void index(final String transformation, final Set<Resource> uris) throws InterruptedException {
94  
95  		log.info("Using transformation:\n{}", transformation);
96  
97  		dataset.begin(READ);
98  		log.trace("Operating with triples:\n{}", dataset.getDefaultModel());
99  		dataset.end();
100 
101 		log.info("Writing to output location: {}", persister.location());
102 
103 		final IndexRun indexRun = new IndexRun(transformation, uris, JenaBackend.with(dataset))
104 				.threads(numIndexerThreads);
105 
106 		// workflow!
107 		indexRun.andThen(outputStage);
108 		outputStage.andThen(persister);
109 		indexRun.run();
110 		indexRun.shutdown();
111 		outputStage.shutdown();
112 		persister.shutdown();
113 	}

155 	/**
156 	 * @param args
157 	 * @throws ParseException
158 	 * @throws IOException
159 	 * @throws InterruptedException
160 	 */
161 	public static void main(final String[] args) throws ParseException, IOException, InterruptedException {
162 		final Parser cliParser = new BasicParser();
163 		// first check to see if this is an invocation of help
164 		final CommandLine helpCmd = cliParser.parse(helpOptions, args, true);
165 		if (helpCmd.getOptions().length > 0) {
166 			new HelpFormatter().printHelp(helpLine, mainOptions);
167 		} else {
168 			// it's not a help invocation, so actually parse the args
169 			try {
170 				final CommandLine cmd = cliParser.parse(mainOptions, args);
171 
172 				final File uriFile = new File(cmd.getOptionValue(URIS.opt()));
173 				log.info("Using URI file: {}", uriFile.getAbsolutePath());
174 				final Set<Resource> uris = retrieveUrisFromFile(uriFile);
175 
176 				final File transformFile = new File(cmd.getOptionValue(TRANSFORM.opt()));
177 				final String transform = Files.toString(transformFile, UTF_8);
178 
179 				final String outputDirectory = cmd.getOptionValue(OUTPUTDIR.opt());
180 
181 				final Workflow main = new Workflow();
182 				if (cmd.hasOption(CACHE.opt())) {
183 					final String cacheFile = cmd.getOptionValue(CACHE.opt());
184 					log.info("Using cache location: {}", cacheFile);
185 					final Dataset dataset = createDataset(cacheFile);
186 					dataset.begin(READ);
187 					main.dataset = dataset;
188 					dataset.end();
189 				} else {
190 					main.dataset = createDataset();
191 				}
192 				Set<Resource> urisToIndex;
193 				if (!cmd.hasOption(SKIPRETRIEVAL.opt())) {
194 					Integer assemblerThreads = DEFAULT_NUM_THREADS;
195 					if (cmd.hasOption(ASSEMBLERTHREADS.opt())) {
196 						assemblerThreads = parseInt(cmd.getOptionValue(ASSEMBLERTHREADS.opt()));
197 					}
198 					final CacheAssembler assembler = new CacheAssembler(main.dataset).threads(assemblerThreads).uris(
199 							uris);
200 					if (cmd.hasOption(ACCEPT.opt())) {
201 						final String accept = cmd.getOptionValue(ACCEPT.opt());
202 						log.info("Requesting HTTP Content-type: {} for resource retrieval.", accept);
203 						assembler.accepts(accept);
204 					}
205 					main.assembler(assembler);
206 					urisToIndex = main.cache(uris);
207 					log.debug("Successfully retrieved URIs:\n{}", urisToIndex);
208 				} else {
209 					urisToIndex = uris;
210 					if (!cmd.hasOption(CACHE.opt())) {
211 						final String bangLine = repeat("!", 80);
212 						log.warn(bangLine);
213 						log.warn("Operating over empty in-memory cache without retrieval step to fill it!");
214 						log.warn(bangLine);
215 					}
216 				}
217 				if (cmd.hasOption(INDEXINGTHREADS.opt())) {
218 					main.numIndexerThreads = parseInt(cmd.getOptionValue(INDEXINGTHREADS.opt()));
219 				}
220 				main.outputStage(new SolrXMLOutputStage());
221 				main.persister(new FilesystemPersister().location(outputDirectory));
222 				main.index(transform, urisToIndex);
223 			} catch (final ParseException e) {
224 				new HelpFormatter().printHelp(helpLine, mainOptions);
225 				throw e;
226 			}
227 		}
228 	}

230 	private static Set<Resource> retrieveUrisFromFile(final File f) throws IOException {
231 		return new HashSet<>(transform(Files.readLines(f, UTF_8), new Function<String, Resource>() {
232 
233 			@Override
234 			public Resource apply(final String uri) {
235 				return createResource(uri);
236 			}
237 		}));
238 	}
