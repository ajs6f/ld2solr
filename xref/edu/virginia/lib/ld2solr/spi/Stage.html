3   /**
4    * A stage of workflow.
5    * 
6    * @author ajs6f
7    * @param <Produces>
8    *            the type of thing produced by this stage
9    */
10  public interface Stage<Produces> {
11  
12  	/**
13  	 * Assign the next stage of workflow.
14  	 * 
15  	 * @param a
16  	 *            the next {@link Stage} of workflow
17  	 */
18  	public void andThen(Acceptor<Produces, ?> a);
19  
20  	/**
21  	 * Hand a task over to the next stage of workflow.
22  	 * 
23  	 * @param task
24  	 */
25  	public void next(Produces task);
26  
27  	/**
28  	 * Frees any resources associated to this {@link Stage}.
29  	 * 
30  	 * @throws InterruptedException
31  	 */
32  	public void shutdown() throws InterruptedException;
33  
34  	/**
35  	 * A stage of workflow that is not the first and therefore accepts tasks.
36  	 * 
37  	 * @author ajs6f
38  	 * @param <Accepts>
39  	 *            the type of thing accepted by this stage
40  	 * @param <Produces>
41  	 *            the type of thing produced by this stage
42  	 */
43  	public interface Acceptor<Accepts, Produces> extends Stage<Produces> {
44  
45  		/**
46  		 * Accept a task or quantum of work.
47  		 * 
48  		 * @param task
49  		 *            the task to perform
50  		 */
51  		public void accept(Accepts task);
52  	}
53  
54  }
