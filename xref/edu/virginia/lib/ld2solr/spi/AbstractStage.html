1   package edu.virginia.lib.ld2solr.spi;

3   import static com.google.common.util.concurrent.MoreExecutors.listeningDecorator;
4   import static java.lang.Integer.MAX_VALUE;
5   import static java.util.concurrent.Executors.newFixedThreadPool;
6   import static java.util.concurrent.TimeUnit.SECONDS;
7   
8   import com.google.common.util.concurrent.ListeningExecutorService;
9   
10  /**
11   * @author ajs6f
12   * 
13   */
14  public abstract class AbstractStage<T extends AbstractStage<T, Produces>, Produces> implements Stage<Produces> {
15  
16  	public static final Integer DEFAULT_NUM_THREADS = 10;
17  
18  	protected ListeningExecutorService threadpool = listeningDecorator(newFixedThreadPool(DEFAULT_NUM_THREADS));
19  
20  	protected Acceptor<Produces, ?> nextStage;
21  
22  	@Override
23  	public void andThen(final Acceptor<Produces, ?> a) {
24  		this.nextStage = a;
25  	}
26  
27  	/*
28  	 * (non-Javadoc)
29  	 * 
30  	 * @see edu.virginia.lib.ld2solr.spi.Stage#next(java.lang.Object)
31  	 */
32  	@Override
33  	public void next(final Produces task) {
34  		nextStage.accept(task);
35  	}
36  
37  	/**
38  	 * @return a {@link ListeningExecutorService} that should be used for work
39  	 *         associated with tasks transiting this stage
40  	 */
41  	protected ListeningExecutorService threadpool() {
42  		return threadpool;
43  	}
44  
45  	@Override
46  	public void shutdown() throws InterruptedException {
47  		this.threadpool.shutdown();
48  		this.threadpool.awaitTermination(MAX_VALUE, SECONDS);
49  	}
50  
51  	/**
52  	 * Alter the number of threads being used in this {@link Stage}.
53  	 * 
54  	 * @param numThreads
55  	 *            the number of threads to use
56  	 * @throws InterruptedException
57  	 */
58  	public T threads(final Integer numThreads) throws InterruptedException {
59  		shutdown();
60  		threadpool = listeningDecorator(newFixedThreadPool(numThreads));
61  		@SuppressWarnings("unchecked")
62  		final T t = (T) this;
63  		return t;
64  	}
65  }
