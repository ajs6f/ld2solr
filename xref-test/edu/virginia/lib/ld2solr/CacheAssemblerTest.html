24  public class CacheAssemblerTest extends TestHelper {
25  
26  	private DatasetCacheAssembler testAssembler;
27  
28  	private Dataset dataset;
29  
30  	private static final Logger log = getLogger(CacheAssemblerTest.class);
31  
32  	@Before
33  	public void setUp() throws InterruptedException {
34  		dataset = createDataset();
35  		testAssembler = new DatasetCacheAssembler().cache(dataset).threads(3);
36  	}
37  
38  	@Test
39  	public void testAccumulation() {
40  		final Set<Resource> successfullyRetrievedUris = testAssembler.load(uris);
41  		assertEquals("Did not retrieve all resources successfully!", uris, successfullyRetrievedUris);
42  		dataset.begin(READ);
43  		log.debug("Retrieved triples: {}", dataset.getDefaultModel());
44  		for (final Resource uri : uris)
45  			assertTrue("Did not find an appropriate subject " + uri + " in triplestore!", dataset.getDefaultModel()
46  					.containsResource(uri));
47  		dataset.end();
48  	}
49  
50  	@Test
51  	public void testAccumulationWithEmptyResource() {
52  		final Set<Resource> urisWithEmpty = new HashSet<>(uris);
53  		urisWithEmpty.add(createResource(uriBase + "empty"));
54  		final Set<Resource> successfullyRetrievedUris = testAssembler.load(urisWithEmpty);
55  		assertTrue("Did not retrieve all resources successfully!", successfullyRetrievedUris.containsAll(uris));
56  		dataset.begin(READ);
57  		log.debug("Retrieved triples: {}", dataset.getDefaultModel());
58  		for (final Resource uri : uris)
59  			assertTrue("Did not find an appropriate subject " + uri + " in triplestore!", dataset.getDefaultModel()
60  					.containsResource(uri));
61  		dataset.end();
62  	}
63  
64  	@Test
65  	public void testAccumulationWithProblem() {
66  		final Set<Resource> urisWithExtra = new HashSet<>(uris);
67  		final Set<Resource> badUris = singleton(createResource());
68  		urisWithExtra.addAll(badUris);
69  		testAssembler = new DatasetCacheAssembler().cache(dataset);
70  		final Set<Resource> successfulUris = testAssembler.load(urisWithExtra);
71  		assertEquals("Didn't find the appropriate resource failing to be retrieved!", badUris,
72  				difference(urisWithExtra, successfulUris));
73  	}
