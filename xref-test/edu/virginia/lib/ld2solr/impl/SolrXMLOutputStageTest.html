23  public class SolrXMLOutputStageTest {
24  
25  	private SolrXMLOutputStage testStage;
26  
27  	private TestSink testSink;
28  
29  	private static final String testFieldValue = "testFieldValue";
30  
31  	private NamedFields fields;
32  
33  	private static final Logger log = getLogger(SolrXMLOutputStageTest.class);
34  
35  	@Before
36  	public void setUp() {
37  		testStage = new SolrXMLOutputStage();
38  		testSink = new TestSink();
39  		testStage.andThen(testSink);
40  		fields = new NamedFields(ImmutableMap.<String, Collection<?>> of(ID_FIELD, asList(testFieldValue)));
41  	}
42  
43  	/**
44  	 * 
45  	 */
46  	@Test
47  	public void testAccept() {
48  		testStage.accept(fields);
49  		assertTrue("Didn't find our record in output from test stage!",
50  				Iterables.any(testSink.accepted(), new Predicate<OutputRecord>() {
51  
52  					@Override
53  					public boolean apply(final OutputRecord record) {
54  
55  						return record.id().equals(fields.id());
56  					}
57  				}));
58  		final String outputBytes = new String(testSink.accepted().iterator().next().record());
59  		log.debug("Retrieved record: {}", outputBytes);
60  		assertTrue("Failed to find test value in output record!", outputBytes.contains(testFieldValue));
61  	}
62  
63  	@Test
64  	public void testWrap() throws InterruptedException {
65  		testStage = new SolrXMLOutputStage().threads(2);
66  		final SolrInputDocument result = SolrXMLOutputStage.wrap.apply(fields);
67  		assertTrue("Should have found test field value under test field name in Solr document!",
68  				result.getFieldValues(ID_FIELD).contains(testFieldValue));
69  
70  	}
