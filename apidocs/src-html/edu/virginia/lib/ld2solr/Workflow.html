051/**
052 * Assembles and operates an indexing workflow from SPI implementations, as well
053 * as CLI support therefor.
054 * 
055 * @author ajs6f
056 * 
057 */
058public class Workflow {
059
060        /*
061         * because LDPath uses java.util.ServiceLoader to search the context
062         * classpath, we need to explicitly call out LDPath here to make sure it
063         * gets initialized properly
064         */
065        @SuppressWarnings("unused")
066        private final LDPath<String> testTransform = new LDPath<>(null);
067
068        private Dataset dataset;
069
070        private CacheAssembler cacheAssembler = null;
071
072        private RecordPersister persister;
073
074        private OutputStage outputStage = null;
075
076        private Integer numIndexerThreads = ThreadedStage.DEFAULT_NUM_THREADS;
077
078        private static final Logger log = getLogger(Workflow.class);
079
080        public Set<Resource> fullRun(final String transformation, final Set<Resource> uris) throws InterruptedException {
081
082                log.info("Using transformation:\n{}", transformation);
083
084                Set<Resource> successfullyRetrieved = new HashSet<>(uris.size());
085                // first, we may need to assemble the cache of RDF
086                if (cacheAssembler != null) {
087                        successfullyRetrieved.addAll(cacheAssembler.call());
088                        final Set<Resource> failures = difference(uris, successfullyRetrieved);
089                        if (failures.size() > 0) {
090                                log.error("Failed to retrieve some resources!");
091                                for (final Resource failure : failures) {
092                                        log.warn("Resource: {} could not be retrieved!", failure);
093                                }
094                        }
095                        cacheAssembler.shutdown();
096                } else {
097                        // assume fully cached data
098                        successfullyRetrieved = uris;
099                }
100                dataset.begin(READ);
101                log.trace("Operating with triples:\n{}", dataset.getDefaultModel());
102                dataset.end();
103
104                log.info("Writing to output location: {}", persister.location());
105
106                final IndexRun indexRun = new IndexRun(transformation, successfullyRetrieved, JenaBackend.with(dataset))
107                                .threads(numIndexerThreads);
108
109                // workflow!
110                indexRun.andThen(outputStage);
111                outputStage.andThen(persister);
112                indexRun.run();
113                indexRun.shutdown();
114                outputStage.shutdown();
115                persister.shutdown();
116
117                return successfullyRetrieved;
118        }
119

122         *            the {@link Dataset} to use underneath the RDF cache

160        /**
161         * @param args
162         * @throws ParseException
163         * @throws IOException
164         * @throws InterruptedException
165         */
166        public static void main(final String[] args) throws ParseException, IOException, InterruptedException {
167                final Parser cliParser = new BasicParser();
168                // first check to see if this is an invocation of help
169                final CommandLine helpCmd = cliParser.parse(helpOptions, args, true);
170                if (helpCmd.getOptions().length > 0) {
171                        new HelpFormatter().printHelp(helpLine, mainOptions);
172                } else {
173                        // it's not a help invocation, so actually parse the args
174                        try {
175                                final CommandLine cmd = cliParser.parse(mainOptions, args);
176
177                                final File uriFile = new File(cmd.getOptionValue(URIS.opt()));
178                                log.info("Using URI file: {}", uriFile.getAbsolutePath());
179                                final Set<Resource> uris = retrieveUrisFromFile(uriFile);
180
181                                final File transformFile = new File(cmd.getOptionValue(TRANSFORM.opt()));
182                                final String transform = Files.toString(transformFile, UTF_8);
183
184                                final String outputDirectory = cmd.getOptionValue(OUTPUTDIR.opt());
185
186                                final Workflow main = new Workflow();
187                                if (cmd.hasOption(CACHE.opt())) {
188                                        final String cacheFile = cmd.getOptionValue(CACHE.opt());
189                                        log.info("Using cache location: {}", cacheFile);
190                                        final Dataset dataset = createDataset(cacheFile);
191                                        dataset.begin(READ);
192                                        main.dataset = dataset;
193                                        dataset.end();
194                                } else {
195                                        main.dataset = createDataset();
196                                }
197
198                                if (!cmd.hasOption(SKIPRETRIEVAL.opt())) {
199                                        Integer assemblerThreads = DEFAULT_NUM_THREADS;
200                                        if (cmd.hasOption(ASSEMBLERTHREADS.opt())) {
201                                                assemblerThreads = parseInt(cmd.getOptionValue(ASSEMBLERTHREADS.opt()));
202                                        }
203                                        final CacheAssembler assembler = new CacheAssembler(main.dataset).threads(assemblerThreads).uris(
204                                                        uris);
205                                        if (cmd.hasOption(ACCEPT.opt())) {
206                                                final String accept = cmd.getOptionValue(ACCEPT.opt());
207                                                log.info("Requesting HTTP Content-type: {} for resource retrieval.", accept);
208                                                assembler.accepts(accept);
209                                        }
210                                        main.assembler(assembler);
211                                } else if (!cmd.hasOption(CACHE.opt())) {
212                                        final String bangLine = repeat("!", 80);
213                                        log.warn(bangLine);
214                                        log.warn("Operating over empty in-memory cache without retrieval step to fill it!");
215                                        log.warn(bangLine);
216                                }
217                                if (cmd.hasOption(INDEXINGTHREADS.opt())) {
218                                        main.numIndexerThreads = parseInt(cmd.getOptionValue(INDEXINGTHREADS.opt()));
219                                }
220                                main.outputStage(new SolrXMLOutputStage());
221                                main.persister(new FilesystemPersister().location(outputDirectory));
222                                final Set<Resource> successfullyRetrieved = main.fullRun(transform, uris);
223                                log.debug("Successfully retrieved URIs: {}", successfullyRetrieved);
224                        } catch (final ParseException e) {
225                                new HelpFormatter().printHelp(helpLine, mainOptions);
226                                throw e;
227                        }
228                }
229        }
230
231        private static Set<Resource> retrieveUrisFromFile(final File f) throws IOException {
232                return new HashSet<>(transform(Files.readLines(f, UTF_8), new Function<String, Resource>() {
233
234                        @Override
235                        public Resource apply(final String uri) {
236                                return createResource(uri);
237                        }
238                }));
239        }
