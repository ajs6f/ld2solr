042/**
043 * Assembles and operates an indexing workflow from SPI implementations, as well
044 * as CLI support therefor.
045 * 
046 * @author ajs6f
047 * 
048 */
049public class Workflow {

051        /*
052         * because LDPath uses java.util.ServiceLoader to search the context
053         * classpath, we need to explicitly call out LDPath here to make sure it
054         * gets initialized properly
055         */
056        @SuppressWarnings("unused")
057        private final LDPath<String> testTransform = new LDPath<>(null);
058
059        private Dataset dataset;
060
061        private CacheAssembler cacheAssembler = null;
062
063        private RecordPersister persister;
064
065        private OutputStage outputStage = null;
066
067        private Integer numIndexerThreads = AbstractStage.DEFAULT_NUM_THREADS;
068
069        private static final Logger log = getLogger(Workflow.class);
070
071        public Set<Resource> fullRun(final String transformation, final Set<Resource> uris) throws InterruptedException {
072
073                Set<Resource> successfullyRetrieved = new HashSet<>(uris.size());
074                // first, we may need to assemble the cache of RDF
075                if (cacheAssembler != null) {
076                        successfullyRetrieved.addAll(cacheAssembler.call());
077                        final Set<Resource> failures = difference(uris, successfullyRetrieved);
078                        if (failures.size() > 0) {
079                                log.error("Failed to retrieve some resources!");
080                                for (final Resource failure : failures) {
081                                        log.warn("Resource: {} could not be retrieved!", failure);
082                                }
083                        }
084                        cacheAssembler.shutdown();
085                } else {
086                        // assume fully cached data
087                        successfullyRetrieved = uris;
088                }
089                dataset.begin(READ);
090                // log.trace("Operating with triples:\n{}", dataset.getDefaultModel());
091                dataset.end();
092
093                log.info("Writing to output location: {}", persister.location());
094
095                final IndexRun indexRun = new IndexRun(transformation, successfullyRetrieved, JenaBackend.with(dataset))
096                                .threads(numIndexerThreads);
097
098                // workflow!
099                indexRun.andThen(outputStage);
100                outputStage.andThen(persister);
101                indexRun.run();
102                indexRun.shutdown();
103                outputStage.shutdown();
104                persister.shutdown();
105
106                return successfullyRetrieved;
107        }

149        /**
150         * @param args
151         * @throws ParseException
152         * @throws IOException
153         * @throws InterruptedException
154         */
155        public static void main(final String[] args) throws ParseException, IOException, InterruptedException {
156                final CommandLine cmd = new BasicParser().parse(getOptions(), args);
157                if (cmd.hasOption('h')) {
158                        new HelpFormatter().printHelp("ld2solr -t transform-file -o output-dir -u input-uris", getOptions());
159                } else {
160                        if (!cmd.hasOption('u') || !cmd.hasOption('t') || !cmd.hasOption('o')) {
161                                new HelpFormatter().printHelp("ld2solr -t transform-file -o output-dir -u input-uris", getOptions());
162                                throw new IllegalArgumentException(
163                                                "This utility requires a list of URIs to index, a transform to use for indexing and a place to put the index records!\n");
164                        }
165                        final String separator = cmd.hasOption('s') ? cmd.getOptionValue('s') : "\n";
166                        final File uriFile = new File(cmd.getOptionValue('u'));
167                        log.info("Using URI file: {}", uriFile.getAbsolutePath());
168                        final Set<Resource> uris = new HashSet<>(transform(asList(Files.toString(uriFile, UTF_8).split(separator)),
169                                        string2uri));
170                        final File transformFile = new File(cmd.getOptionValue("transform"));
171                        log.info("Using transform file: {}", transformFile.getAbsolutePath());
172                        final String transform = Files.toString(transformFile, UTF_8);
173                        final String outputDirectory = cmd.getOptionValue('o');
174                        log.info("Using output directory: {}", outputDirectory);
175                        final Workflow main = new Workflow();
176                        if (cmd.hasOption('c')) {
177                                final String cacheFile = cmd.getOptionValue('c');
178                                log.info("Using cache location: {}", cacheFile);
179                                final Dataset dataset = createDataset(cacheFile);
180                                dataset.begin(READ);
181                                main.dataset = dataset;
182                                dataset.end();
183                        } else {
184                                main.dataset = createDataset();
185                        }
186                        if (!cmd.hasOption("skip-retrieval")) {
187                                Integer assemblerThreads = AbstractStage.DEFAULT_NUM_THREADS;
188                                if (cmd.hasOption("assembler-threads")) {
189                                        assemblerThreads = parseInt(cmd.getOptionValue("assembler-threads"));
190                                }
191                                final CacheAssembler assembler = new CacheAssembler(main.dataset).threads(assemblerThreads).uris(uris);
192                                if (cmd.hasOption('a')) {
193                                        final String accept = cmd.getOptionValue('a');
194                                        log.info("Requesting content-type: {} for resource retrieval.", accept);
195                                        assembler.accepts(accept);
196                                }
197                                main.assembler(assembler);
198                        } else if (!cmd.hasOption('c')) {
199                                final String bangLine = repeat("!", 80);
200                                log.warn(bangLine);
201                                log.warn("Operating over empty in-memory cache without retrieval step to fill it!");
202                                log.warn(bangLine);
203                        }
204                        if (cmd.hasOption("indexing-threads")) {
205                                main.numIndexerThreads = parseInt(cmd.getOptionValue("indexing-threads"));
206                        }
207                        main.outputStage(new SolrXMLOutputStage());
208                        main.persister(new FilesystemPersister().location(outputDirectory));
209                        final Set<Resource> successfullyRetrieved = main.fullRun(transform, uris);
210                        log.debug("Successfully retrieved URIs: {}", successfullyRetrieved);
211                }
212        }

214        private static Options getOptions() {
215                return new Options()
216                                .addOption("u", "uris", true, "(Required) A file or pipe with a list of URIs to index.")
217                                .addOption("o", "output-dir", true, "(Required) Location into which to place output files.")
218                                .addOption("t", "transform", true,
219                                                "(Required) Location of LDPath transform with which to create index records. ")
220                                .addOption("c", "cache", true,
221                                                "Location of persistent triple cache. (Defaults to in-memory only operation.)")
222                                .addOption("a", "accept", true, "HTTP 'Accept:' header to use. (Defaults to none.)")
223                                .addOption(
224                                                "sr",
225                                                "skip-retrieval",
226                                                false,
227                                                "Should retrieval and caching of Linked Data resources before indexing stages be skipped?"
228                                                                + "If set, option for persistent triple cache must be supplied or "
229                                                                + "indexing stages will operate over an empty cache.")
230                                .addOption("s", "separator", true, "Separator between input URIs. (Defaults to \\n.)")
231                                .addOption(
232                                                new Option(null, "assembler-threads", true,
233                                                                "The number of threads to use for RDF cache accumulation. (Defaults to "
234                                                                                + AbstractStage.DEFAULT_NUM_THREADS + ".)"))
235                                .addOption(
236                                                new Option(null, "indexing-threads", true,
237                                                                "The number of threads to use for indexing operation. (Defaults to "
238                                                                                + AbstractStage.DEFAULT_NUM_THREADS + ".)"))
239                                .addOption("h", "help", false, "Print this help message.");
240        }

242        private static Function<String, Resource> string2uri = new Function<String, Resource>() {
243
244                @Override
245                public Resource apply(final String uri) {
246                        return createResource(uri);
247                }
248        };
