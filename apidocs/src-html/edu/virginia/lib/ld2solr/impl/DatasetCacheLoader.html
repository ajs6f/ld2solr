001package edu.virginia.lib.ld2solr.impl;
002
003import static com.google.common.util.concurrent.Futures.addCallback;
004import static com.hp.hpl.jena.query.ReadWrite.WRITE;
005import static org.slf4j.LoggerFactory.getLogger;
006
007import org.slf4j.Logger;
008
009import com.google.common.util.concurrent.FutureCallback;
010import com.google.common.util.concurrent.ListenableFuture;
011import com.hp.hpl.jena.query.Dataset;
012import com.hp.hpl.jena.rdf.model.Model;
013
014import edu.virginia.lib.ld2solr.spi.AbstractStage;
015import edu.virginia.lib.ld2solr.spi.CacheLoader;
016
017/**
018 * A {@link CacheLoader} that loads RDF into a Jena {@link Dataset}.
019 * 
020 * @author ajs6f
021 * 
022 */
023public class DatasetCacheLoader extends AbstractStage<DatasetCacheLoader, Void> implements CacheLoader {
024
025        private Dataset dataset;
026
027        private static final Logger log = getLogger(CacheAssembler.class);
028
029        @Override
030        public void accept(final Model additions) {
031                final ListenableFuture<?> task = threadpool().submit(new Runnable() {
032                        @Override
033                        public void run() {
034                                dataset.begin(WRITE);
035                                try {
036                                        dataset.getDefaultModel().add(additions);
037                                        dataset.commit();
038                                } finally {
039                                        dataset.end();
040                                }
041                        }
042                });
043                addCallback(task, new ReportCallback(additions));
044        }
045
046        private static class ReportCallback implements FutureCallback<Object> {
047
048                private final Model model;
049
050                public ReportCallback(final Model m) {
051                        this.model = m;
052                }
053
054                @Override
055                public void onSuccess(final Object result) {
056                        log.debug("Added {} triples to cache...", model.size());
057                        log.trace("Added triples: {}", model);
058                }
059
060                @Override
061                public void onFailure(final Throwable t) {
062                        log.error("Error assembling triples to add to cache!");
063                        log.error("Exception: ", t);
064                }
065
066        }
067
068        /**
069         * @param d
070         *            the {@link Dataset} with which to back this loader
071         * @return this {@link DatasetCacheLoader} for continued operation
072         */
073        public DatasetCacheLoader dataset(final Dataset d) {
074                this.dataset = d;
075                return this;
076        }
077}
