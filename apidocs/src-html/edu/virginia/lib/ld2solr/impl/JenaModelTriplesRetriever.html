001package edu.virginia.lib.ld2solr.impl;
002
003import static com.hp.hpl.jena.rdf.model.ModelFactory.createDefaultModel;
004import static org.slf4j.LoggerFactory.getLogger;
005
006import java.io.Closeable;
007import java.io.IOException;
008import java.util.concurrent.Callable;
009
010import org.apache.any23.Any23;
011import org.apache.any23.ExtractionReport;
012import org.apache.any23.extractor.ExtractionContext;
013import org.apache.any23.extractor.ExtractionException;
014import org.apache.any23.extractor.Extractor;
015import org.apache.any23.extractor.IssueReport.Issue;
016import org.apache.any23.writer.TripleHandler;
017import org.apache.http.client.HttpClient;
018import org.apache.http.client.methods.HttpGet;
019import org.apache.http.impl.client.DefaultHttpClient;
020import org.apache.http.impl.conn.PoolingClientConnectionManager;
021import org.apache.http.util.EntityUtils;
022import org.openrdf.model.URI;
023import org.openrdf.model.Value;
024import org.slf4j.Logger;
025
026import com.hp.hpl.jena.rdf.model.Literal;
027import com.hp.hpl.jena.rdf.model.Model;
028import com.hp.hpl.jena.rdf.model.Property;
029import com.hp.hpl.jena.rdf.model.Resource;
030
031import edu.virginia.lib.ld2solr.spi.TriplesRetriever;
032
033/**
034 * A {@link TriplesRetriever} that returns its retrieved triples in an in-memory
035 * Jena {@link Model}.
036 * 
037 * @author ajs6f
038 * 
039 */
040public class JenaModelTriplesRetriever implements TriplesRetriever {
041
042        private static final String DEFAULT_USER_AGENT = "UVa Library Linked Data indexing engine";
043
044        private final Any23 extractor = new Any23();
045
046        private static final HttpClient client = new DefaultHttpClient(new PoolingClientConnectionManager());
047
048        private final String accept;
049
050        private static final Logger log = getLogger(JenaModelTriplesRetriever.class);
051
052        public JenaModelTriplesRetriever(final String a) {
053                this.accept = a;
054                this.extractor.setHTTPUserAgent(DEFAULT_USER_AGENT);
055        }
056
057        /*
058         * (non-Javadoc)
059         * 
060         * @see
061         * edu.virginia.lib.ld2solr.impl.TriplesRetriever#load(com.hp.hpl.jena.rdf
062         * .model.Resource)
063         */
064        @Override
065        public Callable<Model> apply(final Resource uri) {
066                return new Callable<Model>() {
067
068                        @Override
069                        public Model call() throws IOException, ExtractionException {
070                                final String resource = uri.getURI();
071                                log.debug("Retrieving from URI: {}", resource);
072                                final HttpGet get = new HttpGet(resource);
073                                if (accept != null) {
074                                        get.setHeader("Accept", accept);
075                                }
076                                final String rdf = EntityUtils.toString(client.execute(get).getEntity());
077                                log.trace("Retrieved from URI: {} RDF:\n{}", resource, rdf);
078                                final Model model = createDefaultModel();
079                                try (final TriplesIntoModel tripleRecorder = new TriplesIntoModel(model);) {
080                                        final ExtractionReport report = extractor.extract(rdf, resource, tripleRecorder);
081                                        if (log.isDebugEnabled()) {
082                                                for (final Extractor<?> extractor : report.getMatchingExtractors()) {
083                                                        for (final Issue issue : report.getExtractorIssues(extractor.getDescription()
084                                                                        .getExtractorName())) {
085                                                                log.debug("Extraction issue in {}: {} at: {}",
086                                                                                new Object[] { resource, issue.getMessage(), issue.getRow() });
087                                                        }
088                                                }
089                                        }
090                                }
091                                return model;
092                        }
093                };
094        }
095
096        /**
097         * Writes extracted triples into a Jena {@link Model}.
098         * 
099         * @author ajs6f
100         * 
101         */
102        private static class TriplesIntoModel implements TripleHandler, Closeable {
103
104                private final Model model;
105
106                public TriplesIntoModel(final Model m) {
107                        this.model = m;
108                }
109
110                @Override
111                public void startDocument(final URI documentURI) {
112                }
113
114                @Override
115                public void openContext(final ExtractionContext context) {
116                }
117
118                @Override
119                public void receiveTriple(final org.openrdf.model.Resource s, final URI p, final Value o, final URI g,
120                                final ExtractionContext context) {
121                        final Property property = model.createProperty(p.stringValue());
122                        final Resource subject = model.createResource(s.stringValue());
123                        if (o instanceof org.openrdf.model.Literal) {
124                                final org.openrdf.model.Literal rawLiteral = (org.openrdf.model.Literal) o;
125                                final URI datatype = rawLiteral.getDatatype();
126                                final Literal literal;
127                                if (datatype != null) {
128                                        literal = model.createTypedLiteral(rawLiteral.getLabel(), datatype.toString());
129                                } else {
130                                        literal = model.createLiteral(rawLiteral.getLabel());
131                                }
132                                model.add(subject, property, literal);
133                        } else {
134                                final Resource resource = model.createResource(o.toString());
135                                model.add(subject, property, resource);
136                        }
137
138                }
139
140                @Override
141                public void receiveNamespace(final String prefix, final String uri, final ExtractionContext context) {
142                        model.setNsPrefix(prefix, uri);
143                }
144
145                @Override
146                public void closeContext(final ExtractionContext context) {
147                }
148
149                @Override
150                public void endDocument(final URI documentURI) {
151                }
152
153                @Override
154                public void setContentLength(final long contentLength) {
155                }
156
157                @Override
158                public void close() {
159                }
160        }
161}
