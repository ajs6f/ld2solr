001package edu.virginia.lib.ld2solr.impl;
002
003import static com.google.common.base.Throwables.propagate;
004import static org.slf4j.LoggerFactory.getLogger;
005
006import java.io.IOException;
007import java.io.Reader;
008import java.io.StringReader;
009import java.util.Collection;
010import java.util.HashMap;
011import java.util.Map;
012
013import org.apache.marmotta.ldpath.LDPath;
014import org.apache.marmotta.ldpath.exception.LDPathParseException;
015import org.apache.marmotta.ldpath.parser.DefaultConfiguration;
016import org.slf4j.Logger;
017
018import com.hp.hpl.jena.rdf.model.RDFNode;
019import com.hp.hpl.jena.rdf.model.Resource;
020
021import edu.virginia.lib.ld2solr.api.NamedFields;
022import edu.virginia.lib.ld2solr.spi.IndexingTransducer;
023
024/**
025 * An {@link IndexingTransducer} that indexes from a {@link JenaBackend} Linked
026 * Data store using an LDPath transformation.
027 * 
028 * @author ajs6f
029 * 
030 */
031public class JenaIndexingTransducer implements IndexingTransducer {
032
033        private final String transformation;
034
035        private final JenaBackend cache;
036
037        private static final Logger log = getLogger(JenaIndexingTransducer.class);
038
039        /**
040         * @param linkedDataCache
041         * @param transformationSource
042         */
043        public JenaIndexingTransducer(final JenaBackend linkedDataCache, final String transformationSource) {
044                this.transformation = transformationSource;
045                this.cache = linkedDataCache;
046        }
047
048        /*
049         * (non-Javadoc)
050         * 
051         * @see com.google.common.base.Function#apply(java.lang.Object)
052         */
053        @Override
054        public NamedFields apply(final Resource uri) {
055                log.debug("Indexing resource: {}", uri);
056                try (Reader transformationReader = new StringReader(transformation);) {
057                        final Map<String, Collection<?>> result = new LDPath<>(cache, new DefaultConfiguration<RDFNode>() {
058                                @Override
059                                public Map<String, String> getNamespaces() {
060                                        final Map<String, String> namespaces = new HashMap<>(super.getNamespaces());
061                                        namespaces.putAll(cache.model().getNsPrefixMap());
062                                        return namespaces;
063                                }
064                        }).programQuery(uri, transformationReader);
065                        log.trace("LDPath transform returned: {}", result);
066                        final NamedFields fields = new NamedFields(result);
067                        // the identifier of the record is the URI of the resource indexed
068                        fields.id(uri.getURI());
069                        log.trace("Created index fields: {}", fields);
070                        return fields;
071                } catch (final LDPathParseException | IOException e) {
072                        throw propagate(e);
073                }
074        }
075}
