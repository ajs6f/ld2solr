001package edu.virginia.lib.ld2solr.impl;
002
003import static com.hp.hpl.jena.query.ReadWrite.READ;
004
005import org.apache.marmotta.ldpath.backend.jena.GenericJenaBackend;
006
007import com.hp.hpl.jena.query.Dataset;
008import com.hp.hpl.jena.rdf.model.Model;
009
010/**
011 * We subclass {@link GenericJenaBackend} in order to expose the Jena
012 * {@link Model} inside it, because we wish to manipulate that model directly.
013 * 
014 * @author ajs6f
015 * 
016 */
017public class JenaBackend extends GenericJenaBackend {
018
019        private final Model model;
020
021        /**
022         * @param m
023         *            the model with which to back this cache.
024         */
025        public JenaBackend(final Model m) {
026                super(m);
027                this.model = m;
028        }
029
030        /**
031         * A factory method to use when operating over a (transactional)
032         * {@link Dataset}. The default model in the {@link Dataset} is used.
033         * 
034         * @param d
035         *            The {@link Dataset} on which to draw.
036         * @return a {@link JenaBackend} backed by d
037         */
038        public static JenaBackend with(final Dataset d) {
039                d.begin(READ);
040                final Model m = d.getDefaultModel();
041                d.end();
042                return new JenaBackend(m);
043        }
044
045        /**
046         * @return the {@link Model} backing this {@link JenaBackend}
047         */
048        public Model model() {
049                return model;
050        }
051}
