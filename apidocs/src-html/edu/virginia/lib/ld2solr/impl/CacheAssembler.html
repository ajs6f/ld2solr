001package edu.virginia.lib.ld2solr.impl;
002
003import static com.google.common.util.concurrent.Futures.addCallback;
004import static com.google.common.util.concurrent.JdkFutureAdapters.listenInPoolThread;
005import static com.hp.hpl.jena.query.ReadWrite.WRITE;
006import static com.hp.hpl.jena.shared.Lock.READ;
007import static org.slf4j.LoggerFactory.getLogger;
008
009import java.util.HashSet;
010import java.util.Set;
011import java.util.concurrent.Callable;
012import java.util.concurrent.CompletionService;
013import java.util.concurrent.ExecutionException;
014import java.util.concurrent.ExecutorCompletionService;
015import java.util.concurrent.Future;
016
017import org.slf4j.Logger;
018
019import com.google.common.util.concurrent.FutureCallback;
020import com.google.common.util.concurrent.ListenableFuture;
021import com.hp.hpl.jena.query.Dataset;
022import com.hp.hpl.jena.rdf.model.Model;
023import com.hp.hpl.jena.rdf.model.Resource;
024
025import edu.virginia.lib.ld2solr.spi.AbstractStage;
026
027/**
028 * @author ajs6f
029 * 
030 */
031public class CacheAssembler extends AbstractStage<CacheAssembler, Void> implements Callable<Set<Resource>> {
032
033        private CompletionService<Model> internalQueue;
034
035        private final Dataset dataset;
036
037        private Set<Resource> successfullyLoadedResources;
038
039        private Set<Resource> uris;
040
041        private String accepts = null;
042
043        private static final Logger log = getLogger(CacheAssembler.class);
044
045        public CacheAssembler(final Dataset d) {
046                this.dataset = d;
047                this.internalQueue = new ExecutorCompletionService<Model>(this.threadpool);
048        }
049
050        /**
051         * @return a set of URIs of successfully loaded resources
052         */
053        @Override
054        public Set<Resource> call() {
055                successfullyLoadedResources = new HashSet<>(uris.size());
056                for (final Resource uri : uris) {
057                        log.debug("Queueing retrieval task for URI: {}...", uri);
058                        final Future<Model> loadFuture = internalQueue.submit(new JenaModelTriplesRetriever(accepts).apply(uri));
059                        final ListenableFuture<Model> loadTask = listenInPoolThread(loadFuture);
060                        addCallback(loadTask, new FutureCallback<Model>() {
061
062                                @Override
063                                public void onSuccess(final Model result) {
064                                        log.debug("Retrieved URI: {} and will add its contents to cache.", uri);
065                                        successfullyLoadedResources.add(uri);
066                                }
067
068                                @Override
069                                public void onFailure(final Throwable t) {
070                                        log.error("Failed to retrieve: {}!", uri);
071                                        log.error("Exception: ", t);
072                                }
073                        });
074                }
075                log.info("Finished queuing retrieval tasks.");
076                // the only purpose of this loop is to ensure that we execute as many
077                // tasks to add triples to the cache as we have executed tasks to
078                // retrieve triples
079                for (@SuppressWarnings("unused")
080                final Resource uri : uris) {
081                        try {
082                                final Model m = internalQueue.take().get();
083                                if (!m.isEmpty()) {
084                                        m.enterCriticalSection(READ);
085                                        log.debug("Adding {} triples to cache...", m.size());
086                                        log.trace("Adding triples: {}", m);
087                                        try {
088                                                dataset.begin(WRITE);
089                                                try {
090                                                        dataset.getDefaultModel().add(m);
091                                                        dataset.commit();
092                                                } catch (final Exception e) {
093                                                        log.error("Error adding triples to cache!");
094                                                        log.error("Triples: ", m);
095                                                        log.error("Exception: ", e);
096                                                } finally {
097                                                        dataset.end();
098                                                }
099                                        } finally {
100                                                m.leaveCriticalSection();
101                                                m.close();
102                                        }
103                                }
104                        } catch (InterruptedException | ExecutionException e) {
105                                log.error("Error assembling triples to add to cache!");
106                                log.error("Exception: ", e);
107                        }
108                }
109                return successfullyLoadedResources;
110        }
111
112        /**
113         * @param u
114         *            the URIs to retrieve
115         * @return this {@link CacheAssembler} for further operation
116         */
117        public CacheAssembler uris(final Set<Resource> u) {
118                this.uris = u;
119                return this;
120        }
121
122        /*
123         * (non-Javadoc)
124         * 
125         * @see
126         * edu.virginia.lib.ld2solr.spi.AbstractStage#threads(java.lang.Integer)
127         */
128        @Override
129        public CacheAssembler threads(final Integer numThreads) throws InterruptedException {
130                super.threads(numThreads);
131                this.internalQueue = new ExecutorCompletionService<Model>(this.threadpool);
132                return this;
133        }
134
135        /**
136         * @param accepts
137         *            the HTTP Accepts header to use in retrieving Linked Data
138         *            resources
139         * @return this {@link CacheAssembler} for further operation
140         */
141        public CacheAssembler accepts(final String accepts) {
142                this.accepts = accepts;
143                return this;
144        }
145}
