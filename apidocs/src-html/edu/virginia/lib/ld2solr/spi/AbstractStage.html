001package edu.virginia.lib.ld2solr.spi;
002
003import static com.google.common.util.concurrent.MoreExecutors.listeningDecorator;
004import static java.lang.Integer.MAX_VALUE;
005import static java.util.concurrent.Executors.newFixedThreadPool;
006import static java.util.concurrent.TimeUnit.SECONDS;
007
008import com.google.common.util.concurrent.ListeningExecutorService;
009
010/**
011 * @author ajs6f
012 * 
013 */
014public abstract class AbstractStage<T extends AbstractStage<T, Produces>, Produces> implements Stage<Produces> {
015
016        public static final Integer DEFAULT_NUM_THREADS = 10;
017
018        protected ListeningExecutorService threadpool = listeningDecorator(newFixedThreadPool(DEFAULT_NUM_THREADS));
019
020        protected Acceptor<Produces, ?> nextStage;
021
022        @Override
023        public void andThen(final Acceptor<Produces, ?> a) {
024                this.nextStage = a;
025        }
026
027        /*
028         * (non-Javadoc)
029         * 
030         * @see edu.virginia.lib.ld2solr.spi.Stage#next(java.lang.Object)
031         */
032        @Override
033        public void next(final Produces task) {
034                nextStage.accept(task);
035        }
036
037        /**
038         * @return a {@link ListeningExecutorService} that should be used for work
039         *         associated with tasks transiting this stage
040         */
041        protected ListeningExecutorService threadpool() {
042                return threadpool;
043        }
044
045        @Override
046        public void shutdown() throws InterruptedException {
047                this.threadpool.shutdown();
048                this.threadpool.awaitTermination(MAX_VALUE, SECONDS);
049        }
050
051        /**
052         * Alter the number of threads being used in this {@link Stage}.
053         * 
054         * @param numThreads
055         *            the number of threads to use
056         * @throws InterruptedException
057         */
058        public T threads(final Integer numThreads) throws InterruptedException {
059                shutdown();
060                threadpool = listeningDecorator(newFixedThreadPool(numThreads));
061                @SuppressWarnings("unchecked")
062                final T t = (T) this;
063                return t;
064        }
065}
