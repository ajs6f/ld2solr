001package edu.virginia.lib.ld2solr;
002
003import static edu.virginia.lib.ld2solr.spi.ThreadedStage.DEFAULT_NUM_THREADS;
004
005import org.apache.commons.cli.Option;
006import org.apache.commons.cli.Options;
007
008/**
009 * Options for use with ld2solr CLI.
010 * 
011 * @author ajs6f
012 * 
013 */
014public enum CLIOption {
015        URIS(new Option("u", "uris", true, "(Required) A file or pipe with a list of URIs to index."), true),
016        OUTPUTDIR(new Option("o", "output-dir", true, "(Required) Location into which to place output files."), true),
017        TRANSFORM(new Option("t", "transform", true,
018                        "(Required) Location of LDPath transform with which to create index records. "), true),
019        CACHE(
020                        new Option("c", "cache", true,
021                                        "Location of persistent triple cache. (Defaults to in-memory only operation.)")),
022        ACCEPT(new Option("a", "accept-header-value", true,
023                        "HTTP 'Accept:' header to use for HTTP-based Linked Data retrieval. (Defaults to none.)")),
024        SKIPRETRIEVAL(new Option("sr", "skip-retrieval", false,
025                        "Should retrieval and caching of Linked Data resources before indexing stages be skipped?"
026                                        + "If set, option for persistent triple cache must be supplied or "
027                                        + "indexing stages will operate over an empty cache.")),
028        ASSEMBLERTHREADS(new Option("at", "assembler-threads", true,
029                        "The number of threads to use for RDF cache accumulation. (Defaults to " + DEFAULT_NUM_THREADS + ".)")),
030        INDEXINGTHREADS(new Option("it", "indexing-threads", true,
031                        "The number of threads to use for indexing operation. (Defaults to " + DEFAULT_NUM_THREADS + ".)"));
032
033        public Option option;
034
035        private CLIOption(final Option o, final Boolean... required) {
036                this.option = o;
037                if (required.length > 0) {
038                        this.option.setRequired(required[0]);
039                } else {
040                        this.option.setRequired(false);
041                }
042        }
043
044        /**
045         * @return the short name of this {@link Option} for use with Apace Commons
046         *         CLI facilities.
047         */
048        public String opt() {
049                return this.option.getOpt();
050        }
051
052        /**
053         * Help line to use for CLI help invocation.
054         */
055        public static String helpLine = "ld2solr -t transform-file -o output-dir -u input-uris";
056
057        /**
058         * {@link Option}s to trigger CLI help invocation.
059         */
060        public static Options helpOptions = new Options().addOption("h", "help", false, "Print this help message.");
061
062        /**
063         * {@link Option}s to use for CLI configuration.
064         */
065        public static Options mainOptions;
066
067        static {
068                mainOptions = new Options();
069                for (final CLIOption option : values()) {
070                        mainOptions.addOption(option.option);
071                }
072        }
073
074}
