001package edu.virginia.lib.ld2solr.impl;
002
003import static edu.virginia.lib.ld2solr.api.NamedFields.ID_FIELD;
004import static java.util.Arrays.asList;
005import static org.junit.Assert.assertTrue;
006import static org.slf4j.LoggerFactory.getLogger;
007
008import java.util.Collection;
009
010import org.apache.solr.common.SolrInputDocument;
011import org.junit.Before;
012import org.junit.Test;
013import org.slf4j.Logger;
014
015import com.google.common.base.Predicate;
016import com.google.common.collect.ImmutableMap;
017import com.google.common.collect.Iterables;
018
019import edu.virginia.lib.ld2solr.api.NamedFields;
020import edu.virginia.lib.ld2solr.api.OutputRecord;
021import edu.virginia.lib.ld2solr.impl.TestAcceptor.TestSink;
022
023public class SolrXMLOutputStageTest {
024
025        private SolrXMLOutputStage testStage;
026
027        private TestSink testSink;
028
029        private static final String testFieldValue = "testFieldValue";
030
031        private NamedFields fields;
032
033        private static final Logger log = getLogger(SolrXMLOutputStageTest.class);
034
035        @Before
036        public void setUp() {
037                testStage = new SolrXMLOutputStage();
038                testSink = new TestSink();
039                testStage.andThen(testSink);
040                fields = new NamedFields(ImmutableMap.<String, Collection<?>> of(ID_FIELD, asList(testFieldValue)));
041        }
042
043        /**
044         * 
045         */
046        @Test
047        public void testAccept() {
048                testStage.accept(fields);
049                assertTrue("Didn't find our record in output from test stage!",
050                                Iterables.any(testSink.accepted(), new Predicate<OutputRecord>() {
051
052                                        @Override
053                                        public boolean apply(final OutputRecord record) {
054
055                                                return record.id().equals(fields.id());
056                                        }
057                                }));
058                final String outputBytes = new String(testSink.accepted().iterator().next().record());
059                log.debug("Retrieved record: {}", outputBytes);
060                assertTrue("Failed to find test value in output record!", outputBytes.contains(testFieldValue));
061        }
062
063        @Test
064        public void testWrap() throws InterruptedException {
065                testStage = new SolrXMLOutputStage().threads(2);
066                final SolrInputDocument result = SolrXMLOutputStage.wrap.apply(fields);
067                assertTrue("Should have found test field value under test field name in Solr document!",
068                                result.getFieldValues(ID_FIELD).contains(testFieldValue));
069
070        }
071}
