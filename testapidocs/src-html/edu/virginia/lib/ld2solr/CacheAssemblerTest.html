001package edu.virginia.lib.ld2solr;
002
003import static com.google.common.collect.Sets.difference;
004import static com.hp.hpl.jena.query.ReadWrite.READ;
005import static com.hp.hpl.jena.rdf.model.ResourceFactory.createResource;
006import static com.hp.hpl.jena.tdb.TDBFactory.createDataset;
007import static java.util.Collections.singleton;
008import static org.junit.Assert.assertEquals;
009import static org.junit.Assert.assertTrue;
010import static org.slf4j.LoggerFactory.getLogger;
011
012import java.util.HashSet;
013import java.util.Set;
014
015import org.junit.Before;
016import org.junit.Test;
017import org.slf4j.Logger;
018
019import com.hp.hpl.jena.query.Dataset;
020import com.hp.hpl.jena.rdf.model.Resource;
021
022import edu.virginia.lib.ld2solr.CacheAssembler;
023import edu.virginia.lib.ld2solr.impl.TestHelper;
024
025public class CacheAssemblerTest extends TestHelper {
026
027        private CacheAssembler testAssembler;
028
029        private Dataset dataset;
030
031        private static final Logger log = getLogger(CacheAssemblerTest.class);
032
033        @Before
034        public void setUp() throws InterruptedException {
035                dataset = createDataset();
036                testAssembler = new CacheAssembler(dataset).threads(3);
037                testAssembler.uris(uris);
038        }
039
040        @Test
041        public void testAccumulation() {
042                final Set<Resource> successfullyRetrievedUris = testAssembler.call();
043                assertEquals("Did not retrieve all resources successfully!", uris, successfullyRetrievedUris);
044                dataset.begin(READ);
045                log.debug("Retrieved triples: {}", dataset.getDefaultModel());
046                for (final Resource uri : uris)
047                        assertTrue("Did not find an appropriate subject " + uri + " in triplestore!", dataset.getDefaultModel()
048                                        .containsResource(uri));
049                dataset.end();
050        }
051
052        @Test
053        public void testAccumulationWithEmptyResource() {
054                final Set<Resource> urisWithEmpty = new HashSet<>(uris);
055                urisWithEmpty.add(createResource(uriBase + "empty"));
056                testAssembler.uris(urisWithEmpty);
057                final Set<Resource> successfullyRetrievedUris = testAssembler.call();
058                assertTrue("Did not retrieve all resources successfully!", successfullyRetrievedUris.containsAll(uris));
059                dataset.begin(READ);
060                log.debug("Retrieved triples: {}", dataset.getDefaultModel());
061                for (final Resource uri : uris)
062                        assertTrue("Did not find an appropriate subject " + uri + " in triplestore!", dataset.getDefaultModel()
063                                        .containsResource(uri));
064                dataset.end();
065        }
066
067        @Test
068        public void testAccumulationWithProblem() {
069                final Set<Resource> urisWithExtra = new HashSet<>(uris);
070                final Set<Resource> badUris = singleton(createResource());
071                urisWithExtra.addAll(badUris);
072                testAssembler = new CacheAssembler(dataset).uris(urisWithExtra);
073                final Set<Resource> successfulUris = testAssembler.call();
074                assertEquals("Didn't find the appropriate resource failing to be retrieved!", badUris,
075                                difference(urisWithExtra, successfulUris));
076        }
077
078}
