088        @Test
089        public void testFullRun() throws InterruptedException {
090                log.trace("Entering testFullRun()...");
091                final Set<Resource> successfulUris = testMain.fullRun(transformation, uris);
092                final long startTime = currentTimeMillis();
093                synchronized (testSink) {
094                        while (testSink.accepted().size() < successfulUris.size() && currentTimeMillis() < (startTime + TIMEOUT)) {
095                                testSink.wait(TIMESTEP);
096                        }
097                }
098                for (final Resource uri : uris) {
099                        assertTrue("Failed to find a record for " + uri + " in output records: " + testSink.accepted() + "!",
100                                        Iterables.any(testSink.accepted(), new Predicate<OutputRecord>() {
101
102                                                @Override
103                                                public boolean apply(final OutputRecord record) {
104
105                                                        return record.id().equals(uri.toString());
106                                                }
107                                        }));
108                }
109                log.trace("Leaving testFullRun().");
110        }

112        @Test
113        public void testRunWithUnloadableResources() throws InterruptedException {
114                log.trace("Entering testRunWithUnloadableResources()...");
115                final Set<Resource> urisWithExtra = new HashSet<>(uris);
116                final Set<Resource> badUris = singleton(createResource());
117                urisWithExtra.addAll(badUris);
118                final Set<Resource> successfulUris = testMain.fullRun(transformation, urisWithExtra);
119                final long startTime = currentTimeMillis();
120                synchronized (testSink) {
121                        while ((testSink.accepted().size() < successfulUris.size()) && currentTimeMillis() < (startTime + TIMEOUT)) {
122                                testSink.wait(TIMESTEP);
123                        }
124                }
125                assertEquals("Didn't find the appropriate resource failing to be retrieved!", badUris,
126                                difference(urisWithExtra, successfulUris));
127                log.trace("Leaving testRunWithUnloadableResources().");
128        }

199        @Test
200        public void testMissingRequiredCLIArgs() {
201                Exception e = testMainMethod(new String[] { "-t", "fakeTransform", "-u", "fakeUris" });
202                if (e == null) {
203                        fail("Should have failed to execute Workflow.main without required parameter for output directory!");
204                }
205                e = testMainMethod(new String[] { "-t", "fakeTransform", "-o", "fakeOutputDirectory" });
206                if (e == null) {
207                        fail("Should have failed to execute Workflow.main without required parameter for input URIs!");
208                }
209                e = testMainMethod(new String[] { "-u", "fakeUris", "-o", "fakeOutputDirectory" });
210                if (e == null) {
211                        fail("Should have failed to execute Workflow.main without required parameter for indexing transform!");
212                }
213                e = testMainMethod(new String[] { "-o", "fakeOutputDirectory" });
214                if (e == null) {
215                        fail("Should have failed to execute Workflow.main without required parameters for indexing transform and input URIs!");
216                }
217                e = testMainMethod(new String[] { "-u", "fakeUris" });
218                if (e == null) {
219                        fail("Should have failed to execute Workflow.main without required parameters for indexing transform and output directory!");
220                }
221                e = testMainMethod(new String[] { "-t", "fakeTransform" });
222                if (e == null) {
223                        fail("Should have failed to execute Workflow.main without required parameters for input URIs and output directory!");
224                }
225                e = testMainMethod(new String[] {});
226                if (e == null) {
227                        fail("Should have failed to execute Workflow.main without required parameters for input URIs, indexing transform, and output directory!");
228                }
229        }

231        private String[] createBasicArgsForMainMethodTest() throws IOException {
232                final String outputDir = createTempDir().getAbsolutePath();
233                final File transformFile = createTempFile("transform" + randomUUID(), "");
234                write(transformation, transformFile, UTF_8);
235                final File urisFile = createTempFile("uris" + randomUUID(), "");
236                write(on('\n').join(uris), urisFile, UTF_8);
237                return new String[] { "-o", outputDir, "-t", transformFile.getAbsolutePath(), "-u", urisFile.getAbsolutePath() };
238        }
239
240        private Exception testMainMethod(final String args[]) {
241                try {
242                        new Thread(new Runnable() {
243
244                                @Override
245                                public void run() {
246                                        try {
247                                                main(args);
248                                        } catch (final Exception e) {
249                                                propagate(e);
250                                        }
251
252                                }
253                        }).run();
254                } catch (final Exception e) {
255                        return e;
256                }
257                return null;
258        }
